def R_PT(T_PT):
    R_0 = 100 # in Ohm
    A = 3.93e-3 # in 1/¬∞C
    B = -5.7e-7 # in 1/^C^2

#--------- Hei√üleiter NTC -------------

    R_25 = 1000 # in Ohm

    return R_25 * np.exp(B * (1/(T_NTC) - 1/T_25))
